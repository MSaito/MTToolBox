341  int AlgorithmEquidsitribution<T>::get_equidist_main(int v) {
342  using namespace std;
343  using namespace NTL;
344  int bit_len = v;
345  int pivot_index;
346  int old_pivot = 0;
347 
348  pivot_index = calc_1pos(basis[bit_len]->next);
349  while (!basis[bit_len]->zero) {
350 #if defined(DEBUG)
351  if (pivot_index != calc_1pos(basis[pivot_index]->next)) {
352  cerr << "pivot error 1" << endl;
353  cerr << "pivot_index:" << dec << pivot_index << endl;
354  cerr << "calc_1pos:" << dec
355  << calc_1pos(basis[pivot_index]->next) << endl;
356  cerr << "next:" << hex << basis[pivot_index]->next << endl;
357  throw new std::logic_error("pivot error 1");
358  }
359 #endif
360  // アルゴリズムとして、全部のcount を平均的に大きくしたい。
361  // 従って count の小さい方を変化させたい
362  if (basis[bit_len]->count > basis[pivot_index]->count) {
363  swap(basis[bit_len], basis[pivot_index]);
364  }
365  basis[bit_len]->add(*basis[pivot_index]);
366  // add の結果 next の最後の1 は必ず 0 になる。
367  // 全部0なら次の状態に進める。（内部でcount が大きくなる）
368  if (basis[bit_len]->next == 0) {
369  basis[bit_len]->next_state(bit_len);
370  pivot_index = calc_1pos(basis[bit_len]->next);
371 
372  // 全部0でなければ、pivot_index は小さくなる。
373  // pivot_index が 0 になれば最上位bit のみ1なので
374  // 次の add で全部0になる。
375  } else {
376  old_pivot = pivot_index;
377  pivot_index = calc_1pos(basis[bit_len]->next);
378  if (old_pivot <= pivot_index) {
379  cerr << "pivot error 2" << endl;
380  throw new std::logic_error("pivot error 2");
381  }
382  }
383  }
384 
385  // 計算終了したので最長のベクトルを求める。（長いとはcountが少ないこと）
386  int min_count = basis[0]->count;
387  for (int i = 1; i < bit_len; i++) {
388  if (min_count > basis[i]->count) {
389  min_count = basis[i]->count;
390  }
391  }
392  if (min_count > stateBitSize / bit_len) {
393  cout << "over theoretical bound " << bit_len << endl;
394  for(int i = 0; i < size; i++) {
395  basis[i]->debug_print();
396  }
397  throw new std::logic_error("over theoretical bound");
398  }
399  return min_count;
